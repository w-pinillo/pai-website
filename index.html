# Definición de un Modelo de Red Neuronal Profunda 
import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, Dropout, Flatten
from tensorflow.keras.optimizers import Adam

# Definir el modelo de red neuronal secuencial
model = Sequential([
    # Capa de entrada (ejemplo: 784 características para imágenes 28x28)
    Dense(128, activation='relu', input_shape=(784,)),
    Dropout(0.3),
    # Capa oculta
    Dense(64, activation='relu'),
    Dropout(0.3),
    # Capa de salida (ejemplo: 10 clases para clasificación)
    Dense(10, activation='softmax')
])

# Compilar el modelo
model.compile(optimizer=Adam(learning_rate=0.001),
              loss='sparse_categorical_crossentropy',
              metrics=['accuracy'])

# Mostrar un resumen del modelo
model.summary()
